stretch

Stretch a positioned element in relation to its context.

@mixin stretch(
  $top: 0,
  $right: $top,
  $bottom: $top,
  $left: $right,
  $position: absolute
) { /* ... */ }

stretch-x

@mixin stretch-x(
  $left: 0,
  $right: $left,
  $position: absolute
) { /* ... */ }
